O que é Desenvolvimento Web?

  • Criação de sites e aplicações para a web
  • Front-end (HTML, CSS, JS)
  • Back-end (PHP, Node.js, etc)
  • Responsividade e acessibilidade

HTML - Estrutura

<!DOCTYPE html>
<html>
  <head>
    <title>Meu site</title>
  </head>
  <body>
    <h1>Olá, mundo!</h1>
    <p>Esse é meu primeiro site</p>
  </body>
</html>

Tags Principais

  • <h1>...</h1> - Título
  • <p>...</p> - Parágrafo
  • <a href>...</a> - Link
  • <img src> - Imagem
  • <ul><li> - Lista
  • <section><div> - Seções de conteúdo

CSS - Estilização

body {
  background-color: #f0f0f0;
  font-family: Arial;
}
h1 {
  color: blue;
}

Seletores CSS

  • Por tag: p { ... }
  • Por classe: .destaque { ... }
  • Por id: #principal { ... }

Lógica de Programação

  • Variáveis e Constantes
  • Sequência
  • Decisão (if)
  • Repetição (for, while)
  • Arrays

Exemplo: Estrutura de Decisão

let idade = 18;
if (idade >= 18) {
  console.log("Maior de idade");
} else {
  console.log("Menor de idade");
}

Exemplo: Repetição

for (let i = 0; i < 5; i++) {
  console.log("Número: " + i);
}

Arrays

let frutas = ["maçã", "banana", "laranja"];
for (let fruta of frutas) {
  console.log(fruta);
}

Funções em JavaScript

Funções são blocos de código que podem ser reutilizados.

function saudacao() {
    console.log("Olá, mundo!");
}
saudacao();

Função com Parâmetro


function saudacao(nome) {
    console.log("Olá, " + nome + "!");
}
saudacao("Alessandro");

Função com Retorno


function somar(a, b) {
    return a + b;
}
let resultado = somar(3, 4);
console.log(resultado);

Função sem Retorno

function mostrarMensagem(mensagem) {
    alert(mensagem);
}
mostrarMensagem("Bem-vindo!");

Função Anônima e Arrow Function

// Função anônima
let dobro = function(x) {
return x * 2;
};
  
// Arrow function
let triplo = x => x * 3;
  
console.log(dobro(4));
console.log(triplo(4));

Exemplo Prático: Interação com DOM

<button onclick="mudarTexto()">Clique aqui</button>
<p id="paragrafo">Texto original</p>

<script>
function mudarTexto() {
  document.getElementById("paragrafo").innerText = "Texto alterado!";
}
</script>

Exemplo: Eventos e Input

<input type="text" id="nome" placeholder="Digite seu nome">
<button onclick="saudar()">Saudar</button>

<script>
function saudar() {
  let nome = document.getElementById("nome").value;
  alert("Olá, " + nome + "!");
}
</script>

Exemplo: Contador com DOM

<button onclick="incrementar()">+1</button>
<p id="contador">0</p>

<script>
let count = 0;
function incrementar() {
  count++;
  document.getElementById("contador").innerText = count;
}
</script>
