Estruturas de Decisão

  • if / else
  • switch / case
// if / else
  let nota = 10;

  if (nota >= 7) {
    return 'Aprovado';
  } else {
    return 'Reprovado';
  }

// switch
  let n = 1;

  switch (n) {
    case 1: return 'Domingo';
    case 2: return 'Segunda';
    default: return 'Outro dia';
  }

Estruturas de Repetição

  • for, while, do...while
// for
for (let i = 0; i < 10; i++) {
  console.log(i);
}

// while
let n = 5;
while (n > 0) {
  console.log(n--);
}

// do..while
let k = 0;
do {
  console.log(k);
  k++;
} while (k < 3);

Módulo 3 — Lógica com JS na Prática

Strings e Manipulação

  • Métodos comuns de string
  • Exercícios: inverter palavra, contar vogais
function inverter(str) {
  return str.split('').reverse().join('');
}

function contarVogais(str) {
  return (str.match(/[aeiou]/gi) || []).length;
}

console.log(inverter('javascript'));
console.log(contarVogais('javascript'));

Projetinho 1 — Calculadora

Fluxograma + código

// calculadora simples no prompt
const a = Number(prompt('a:'));
const b = Number(prompt('b:'));
const op = prompt('Operador (+ - * /):');
let res;
switch(op) {
  case '+': res = a + b; break;
  case '-': res = a - b; break;
  case '*': res = a * b; break;
  case '/': res = a / b; break;
  default: res = 'Operador inválido';
}
alert(`Resultado: ${res}`);

Tratamento de Erros

  • try / catch / finally

  JSON.parse('texto invalido');
} catch (err) {
  console.error('Erro ao parsear:', err.message);
} finally {
  console.log('Sempre executa');
}

Projetinho 2 — Jogo de Adivinhação

Fluxograma + código

// jogo simples no prompt
const secreto = Math.floor(Math.random() * 100) + 1;
let tentativa = 0;
let achou = false;

while(!achou) {
  const palpite = Number(prompt('Tente um número (1-100):'));
  tentativa++;
  if (palpite === secreto) { alert(`Acertou em ${tentativa} tentativas!`); achou = true; }
  else if (palpite > secreto) alert('Menor');
  else alert('Maior');
}

Projetinho 3 — To-Do List

Estrutura e código principal

// Esqueleto: criar, remover, marcar
class TodoApp {
  constructor() {
    this.todos = JSON.parse(localStorage.getItem('todos') || '[]');
  }
  add(text) { this.todos.push({text, done:false}); this.save(); }
  toggle(i) { this.todos[i].done = !this.todos[i].done; this.save(); }
  remove(i) { this.todos.splice(i,1); this.save(); }
  save() { localStorage.setItem('todos', JSON.stringify(this.todos)); }
}
