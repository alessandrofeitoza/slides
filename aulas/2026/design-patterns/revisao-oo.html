Paradigma Procedural

Baseado em funções e sequência de execução. 


function calcularTotal(produtos) {
  let total = 0;

  for (const produto of produtos) {
    total += produto.preco;
  }

  return total;
}

const total = calcularTotal([
  { preco: 10 },
  { preco: 20 }
]);

No procedural é muito comum encontrar "classes God", Aquele arquivo que todo o sistema roda nele

Paradigma Orientado a Objetos


class Produto {
  constructor(nome, preco) {
    this.nome = nome;
    this.preco = preco;
  }
}

class Carrinho {
  constructor() {
    this.produtos = [];
  }

  adicionar(produto) {
    this.produtos.push(produto);
  }

  calcularTotal() {
    return this.produtos.reduce((t, p) => {
      return t + p.preco;
    }, 0);
  }
}

Pilar: Encapsulamento


class Conta {
  private float $saldo = 0;

  public function setSaldo(float $valor): void
  {
    $this->saldo = $valor;
  }

  public function getSaldo(): float
  {
    return $this->saldo;
  }
}

Pilar: Encapsulamento


class Conta {
  private float $saldo = 0;

  public function depositar(float $valor): void
  {
    if ($valor <= 0) {
      throw new Exception('Valor inválido');
    }

    $this->saldo += $valor;
  }

  public function obterSaldo(): float
  {
    return $this->saldo;
  }
}

Padrão MVC

  • Model: regras e dados
  • View: interface
  • Controller: fluxo da aplicação

class UsuarioController
{
    public function store(Request $request)
    {
        $usuario = Usuario::create($request->all());

        return response()->json($usuario);
    }
}

Repository Pattern


interface UsuarioRepository
{
    public function salvar(Usuario $usuario): void;
    public function buscarPorId(string $id): Usuario;
}

Termos Importantes

  • Entity: objeto do domínio
  • Persist: salvar dados
  • Hydrate: transformar dados em objeto
  • DTO: transporte de dados
  • Mapper: conversão entre estruturas
  • Aggregate: conjunto consistente do domínio
