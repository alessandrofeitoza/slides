Paradigma Procedural

Baseado em funções e sequência de execução. 


function calcularTotal(produtos) {
  let total = 0;

  for (const produto of produtos) {
    total += produto.preco;
  }

  return total;
}

const total = calcularTotal([
  { preco: 10 },
  { preco: 20 }
]);

No procedural é muito comum encontrar "classes God", Aquele arquivo que todo o sistema roda nele

Paradigma Orientado a Objetos


class Produto {
  constructor(nome, preco) {
    this.nome = nome;
    this.preco = preco;
  }
}

class Carrinho {
  constructor() {
    this.produtos = [];
  }

  adicionar(produto) {
    this.produtos.push(produto);
  }

  calcularTotal() {
    return this.produtos.reduce((t, p) => {
      return t + p.preco;
    }, 0);
  }
}

Exemplo Real Completo


class PedidoController {
  constructor(service) {
    this.service = service;
  }

  finalizar(req) {
    return this.service.finalizar(req.body);
  }
}

class PedidoService {
  constructor(repository, gateway) {
    this.repository = repository;
    this.gateway = gateway;
  }

  finalizar(data) {
    const pedido = new Pedido(data.itens);

    this.gateway.pagar(pedido.total());

    this.repository.salvar(pedido);
  }
}
