Use set() concientemente

Regra 8: Use set() concientemente

<?php
    
class EmailSended
{
    private $numberClicks;

    public function setNumberClicks($newClicks)
    {
        $this->numberClicks = $newClicks;
    }

    public function getNumberClicks()
    {
        return $this->numberClicks;
    }
}
